# Draw a chart to show relation between country,job title and role.
data = {
    'Country': ['USA', 'UK', 'Germany', 'Japan'],
    'Job Title': ['Engineer', 'Manager', 'Scientist', 'Designer'],
    'Role': ['Developer', 'Project Lead', 'Researcher', 'Creative Lead']
}


fig, ax = plt.subplots(figsize=(10, 6))


bar_width = 0.2
index = range(len(data['Country']))

bar1 = ax.bar(index, data['Job Title'], bar_width, label='Job Title')
bar2 = ax.bar([i + bar_width for i in index], data['Role'], bar_width, label='Role')


ax.set_xlabel('Country')
ax.set_ylabel('Count')
ax.set_title('Relation between Country, Job Title, and Role')
ax.set_xticks([i + bar_width/2 for i in index])
ax.set_xticklabels(data['Country'])
ax.legend()

plt.show()

# Draw a chart for top 5 roles in 2023 and work type=Intern and Job Title should be Account Director and company should be below 2M.
data = {
    'Role': ['Role A', 'Role B', 'Role C', 'Role D', 'Role E'],
    'Year': [2023, 2023, 2023, 2023, 2023],
    'Work Type': ['Intern', 'Intern', 'Intern', 'Intern', 'Intern'],
    'Job Title': ['Account Director', 'Account Director', 'Account Director', 'Account Director', 'Account Director'],
    'Company Revenue': ['1M', '1.5M', '1.8M', '2.2M', '1.7M']
}


df = pd.DataFrame(data)

# Filter data based on given conditions
filtered_df = df[
    (df['Year'] == 2023) &
    (df['Work Type'] == 'Intern') &
    (df['Job Title'] == 'Account Director') &
    (df['Company Revenue'] < '2M')
]

# Get top 5 roles
top_roles = filtered_df['Role'].value_counts().head(5)

# Plotting
plt.figure(figsize=(10, 6))
top_roles.plot(kind='bar', color='skyblue')
plt.xlabel('Roles')
plt.ylabel('Frequency')
plt.title('Top 5 Roles in 2023 (Work Type: Intern, Job Title: Account Director, Company Revenue < 2M)')
plt.show()
